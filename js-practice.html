一、变量的定义

// 使用 var（旧方式，函数作用域）
var city = "北京";

// 使用 let（块作用域，推荐）
let name = "小明";
let age  = 18;

// 使用 const（常量，不可重新赋值）
const PI = 3.14159;

// 可以先声明后赋值
let score;
score = 95;

console.log(city, name, age, PI, score);

二、变量类型

// 基本数据类型
let num      = 42;           // Number（数字）
let price    = 9.9;          // Number（浮点）
let text     = "Hello JS";   // String（字符串）
let isOnline = true;         // Boolean（布尔）
let nothing  = null;         // Null（空值）
let undef;                   // Undefined（未定义）

// 引用数据类型
let arr = [1, 2, 3, 4, 5];         // Array（数组）
let obj = { name: "李雷", age: 20 };  // Object（对象）

// 用 typeof 检测类型
console.log(typeof num);       // "number"
console.log(typeof text);      // "string"
console.log(typeof isOnline);  // "boolean"
console.log(typeof nothing);   // "object"（JS历史遗留问题）
console.log(typeof undef);     // "undefined"
console.log(typeof arr);       // "object"
console.log(typeof obj);       // "object"

三、表达式

// 算术表达式
let a = 10, b = 3;
console.log("加法:", a + b);   // 13
console.log("减法:", a - b);   // 7
console.log("乘法:", a * b);   // 30
console.log("除法:", a / b);   // 3.333...
console.log("取余:", a % b);   // 1
console.log("幂次:", a ** b);  // 1000

// 字符串拼接
let firstName = "韩";
let lastName  = "梅梅";
let fullName  = firstName + lastName;
console.log("全名:", fullName);

// 模板字符串（反引号）
let greeting = `你好，${fullName}！你今年 ${18} 岁。`;
console.log(greeting);

// 比较表达式（返回 true / false）
console.log("5 > 3:", 5 > 3);
console.log("5 === '5':", 5 === "5");  // 严格相等，false
console.log("5 == '5':", 5 == "5");   // 宽松相等，true

// 逻辑表达式
let x = true, y = false;
console.log("与 &&:", x && y);  // false
console.log("或 ||:", x || y);  // true
console.log("非 !:", !x);       // false

四、流程控制

// --- if / else if / else ---
let grade = 85;
if (grade >= 90) {
  console.log("优秀");
} else if (grade >= 70) {
  console.log("良好");     // 输出这个
} else if (grade >= 60) {
  console.log("及格");
} else {
  console.log("不及格");
}

// --- switch 语句 ---
let day = 3;
switch (day) {
  case 1: console.log("星期一"); break;
  case 2: console.log("星期二"); break;
  case 3: console.log("星期三"); break;  // 输出这个
  default: console.log("其他");
}

// --- for 循环 ---
console.log("for 循环 1~5：");
for (let i = 1; i <= 5; i++) {
  process.stdout ? process.stdout.write(i + " ") : console.log(i);
}

// --- while 循环 ---
let count = 1;
let sum = 0;
while (count <= 100) {
  sum += count;
  count++;
}
console.log("1+2+...+100 =", sum);  // 5050

// --- for...of 遍历数组 ---
let fruits = ["苹果", "香蕉", "橘子"];
for (let fruit of fruits) {
  console.log("水果:", fruit);
}

// --- for...in 遍历对象 ---
let student = { name: "小红", age: 19, major: "计算机" };
for (let key in student) {
  console.log(key + ":", student[key]);
}

五、函数

// --- 函数声明 ---
function greet(name) {
  return "你好，" + name + "！";
}
console.log(greet("同学"));

// --- 有默认参数的函数 ---
function introduce(name, age = 18) {
  return `我叫${name}，今年${age}岁。`;
}
console.log(introduce("小张"));
console.log(introduce("小李", 22));

// --- 函数表达式 ---
const square = function(n) {
  return n * n;
};
console.log("3的平方:", square(3));

// --- 箭头函数 ---
const add = (a, b) => a + b;
console.log("箭头函数 2+5:", add(2, 5));

// --- 递归函数：计算阶乘 ---
function factorial(n) {
  if (n <= 1) return 1;
  return n * factorial(n - 1);
}
console.log("5的阶乘:", factorial(5));  // 120

// --- 函数作用域演示 ---
let globalVar = "全局变量";
function scopeDemo() {
  let localVar = "局部变量";
  console.log("函数内可以访问:", globalVar);
  console.log("函数内局部变量:", localVar);
}
scopeDemo();
